Example jquery code

This example shows how you can add a new user to an existing account. Notice that you cannot reterieve the password, and that you must include the password twice in the request.

$.ajax({
  type: "POST",
  url: "/accounts/acme/users",
  data: {
    "user[username]": "billy",
    "user[email]": "billy@acme.co.uk",
    "user[role]": "shipper",
    "user[password]": "iloveyodel",
    "user[password_confirmation]": "iloveyodel"
  },
  success: function(data){
    $('#result').empty();
    $('#result').append("<ul id='user'></ul>");
    $('#user').append("<li>username is " + $(data).find('username').text() + "</li>");
    $('#user').append("<li>email is " + $(data).find('email').text() + "</li>");
    $('#user').append("<li>role is " + $(data).find('role').text() + "</li>");
  }
});
