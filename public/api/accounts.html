Example jquery code

This simple example will create a new account

$.ajax({
  type: "POST",
  url: "/accounts",
  data: {
      "account[name]": "fred",
      "account[address]": "123 the street",
      "account[postcode]": "SW17 8LW"
  },
  success: function(data){
    $('#result').empty();
    $('#result').append("<ul id='account_details'></ul>");
    $('#account_details').append("<li>id " + $(data).find('id').text() + "</li>");
    $('#account_details').append("<li>name " + $(data).find('account_name').text() + "</li>");
    $('#account_details').append("<li>address " + $(data).find('address').text() + "</li>");
    $('#account_details').append("<li>postcode " + $(data).find('postcode').text() + "</li>");
  }
});
